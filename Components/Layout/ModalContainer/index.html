ModalContainer

ModalContainer components generates a modal whose content can be customized with the modal layouts.

string
required
actionButtonLabel

boolean | undefined

((e: MouseEvent<HTMLDivElement, MouseEvent>) => void) | undefined

string | undefined

number | undefined

Usage

Import this and the layout component you want use and show them when desired.

import React, { useState } from 'react';
import {
  ModalContainer,
  ModalLayoutInfo
} from 'konstellation-web-components';
function Component() {
  const [showModal, setShowModal] = useState(false);
  return (
    <div>
      <div onCLick={() => setShowModal(true)}>RECORD ACTION!</div>
      { showModal && (
        <ModalContainer
          title="CONFIRMATION"
          actionButtonLabel="RECORD ACTION"
          blocking
          warning
          onAccept={() => setShowModal(false)}
          onCancel={() => setShowModal(false)}
          confirmationTimer={5}
        >
          <ModalLayoutInfo>
            This action will be registered in User Activity
          </ModalLayoutInfo>
        </ModalContainer>
      )}
    </div>
  );
}
