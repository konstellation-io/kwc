# ModalContainer

ModalContainer components generates a modal whose content can be customized with the modal layouts.

## Props

<Props of={ModalContainerPropsComponent} />

## Usage

Import this and the layout component you want use and show them when desired.

```tsx
import React, { useState } from 'react';
import { ModalContainer, ModalLayoutInfo } from 'kwc';

function Component() {
  const [showModal, setShowModal] = useState(false);

  return (
    <div>
      <div onCLick={() => setShowModal(true)}>RECORD ACTION!</div>
      {showModal && (
        <ModalContainer
          title="CONFIRMATION"
          actionButtonLabel="RECORD ACTION"
          actionButtonCancel="GO BACK"
          blocking
          warning
          onAccept={() => setShowModal(false)}
          onCancel={() => setShowModal(false)}
          confirmationTimer={5}
        >
          <ModalLayoutInfo>
            This action will be registered in User Activity
          </ModalLayoutInfo>
        </ModalContainer>
      )}
    </div>
  );
}
```
